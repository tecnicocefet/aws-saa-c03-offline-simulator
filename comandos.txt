import json, glob
from collections import Counter

topics = Counter()
tags = Counter()

for path in glob.glob("banks/*.json"):
    with open(path, "r", encoding="utf-8") as f:
        data = json.load(f)
    for q in data.get("questions", []):
        topics[q.get("topic","geral").lower()] += 1
        for t in q.get("tags", []):
            tags[t.lower()] += 1
